\prob{3.9}{Write down all subsets of each of the following.}

\begin{itemize}
  \item[(a)] $\{1, 2, 3\}$
  \item[(b)] $\{\N, \Q, \R\}$
  \item[(c)] $\{\N, \{\Q, \R \}\}$
  \item[(d)] $\emptyset$
\end{itemize}

\noindent
\Ans[(a)] $\emptyset, \{1\}, \{2\}, \{3\}, \{1, 2\}, \{1, 3\}, \{2, 3\}, \{1, 2, 3\}$ \\
\Ans[(b)] $\emptyset, \{\N\}, \{\Q\}, \{\R\}, \{\N, \Q\}, \{\N, \R\}, \{\Q, \R\}, \{\N, \Q, \R\}$ \\
\Ans[(c)] $\emptyset, \{\N\}, \{\{\Q, \R\}\}, \{\N, \{\Q, \R\}\}$ \\
\Ans[(d)] $\emptyset$
